Subroutine Set_Lateral_Flow_Zero(U,V,W,my_imstart,my_imend) 2,2

      USE Model_dim
      USE MPI_dim

      implicit none

      real U(im,jm,NSL),V(im,jm,NSL),W(im,jm,NSL) !Currents
      integer my_imstart,my_imend,i,j,k,myi

      do j = 1,jm
      myi = 2
      do i = my_imstart, my_imend

      if(i.eq.1.or.i.eq.im.or.j.eq.1.or.j.eq.jm) then
        do k = 1,nz
         U( myi, j, k )   = 0.
         V( myi, j, k )   = 0.
         W( myi, j, k )   = 0.
        enddo
      endif

      myi = myi +1
      enddo
      enddo

      end subroutine Set_Lateral_Flow_Zero

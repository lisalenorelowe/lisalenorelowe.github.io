      Subroutine Is_River_Cell(irv,jrv,icent,jcent,lcent)    2,1
!     Find which river (lcent) if icent,jcent is a river cell
!     Return lcent=-1 if not a river cell

      USE Model_dim

      IMPLICIT NONE
      
      INTEGER, INTENT(IN) :: icent, jcent  ! grid cell coordinates of the
                                           ! 1D water column 
      INTEGER, INTENT(IN) :: irv(nr),jrv(nr) 
                                           ! i and j coord of the grid 
                                           ! cell where the mouth each
                                           ! river is located

      INTEGER, INTENT(OUT) :: lcent        ! Number of river, if river cell 
      INTEGER i

      lcent = -1

      do i=1,nr
         if(irv(i).eq.icent.and.jrv(i).eq.jcent) then
           lcent=i
           exit
         endif
      enddo

      Return
      END Subroutine Is_River_Cell

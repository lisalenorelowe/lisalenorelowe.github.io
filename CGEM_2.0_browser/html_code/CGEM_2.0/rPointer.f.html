!------------------------------------------------------------------

       Subroutine  rPointer( nko, no, io, jo, ko ) 2
!    Written by  ::   D.S.Ko  July 1, 1997
!    Modified by ::   Barry Herchenroder/EMVL, April 2010
!------------------------------------------------------------------
      IMPLICIT NONE  
!-----------------------------------------------
! Declare variables 
!-----------------------------------------------
!---------------------------------------------------
! Variables coming through the subroutine interface.
!---------------------------------------------------
      INTEGER,                INTENT(IN)  :: nko 
          
      INTEGER,                INTENT(IN)  :: no       

      INTEGER, DIMENSION(no), INTENT(OUT) :: io, jo, ko

!---------------------------
! local integer scalar variables:
!---------------------------
      INTEGER  imr
      INTEGER  jmr
      INTEGER  nkor 
      INTEGER  nor           

!---------------------------
! character variables:
!---------------------------     
      
      character  fn*72
!-----------------------------------------------          
!     --------------
!     Input Pointers
!     --------------
      fn = TRIM(HYDRODIR//'Pointer.dat')      
      
      open (30,err=8,file=fn,status='old')
      read (30,*    ,err=9) nor,imr,jmr,nkor      
      read (30,*    ,err=9) io,jo,ko    
                                       
      close(30)

#ifdef DEBUG
      write(6,100) no,nor,nko,nkor
      WRITE(6, "(' ')")
      WRITE(6, "(' In subroutine rPointer: ')")
      WRITE(6, "(' ')")       
      WRITE(6, "('no   = ' I8 )") no  
      WRITE(6, "('nor  = ' I8 )") nor 
      WRITE(6, "('nko  = ' I8 )") nko
      WRITE(6, "('nkor = ' I8 )") nkor                            
      WRITE(6, "(' ')")      
100   format(2i8,2i10)
#endif

      if (nor.ne.no.or.nkor.ne.nko) then
	write (6,*)'* wrong dimensions in Pointer.dat'
        WRITE (6, "(' ')")
        WRITE (6, "('Abort Run. ')")	 	
	stop
      end if
      return

8     write(6,*) '*** In subroutine rPointer, cannot find Pointer.dat'
      WRITE (6, "(' ')")
      WRITE (6, "('Abort Run. ')")
      stop

9     write(6,*) '*** In subroutine rPointer, read error at Pointer.dat'
      WRITE (6, "(' ')")
      WRITE (6, "('Abort Run. ')")
      stop

      END SUBROUTINE rPointer

  SUBROUTINE Nitrification( O2, NH4, KO2, KNH4, nitmax, TEMP, R_11 )      2
  !-------------------------------------------------------------------  
  ! Provides Nitrification rate for the model.
  ! Note that the time units are in days, not seconds. 
  !-------------------------------------------------------------------   
    IMPLICIT NONE
    
    REAL, INTENT(IN) :: O2
    REAL, INTENT(IN) :: NH4
    REAL, INTENT(IN) :: KO2
    REAL, INTENT(IN) :: KNH4
    REAL, INTENT(IN) :: nitmax 
    REAL, INTENT(IN) :: TEMP
    REAL, INTENT(OUT):: R_11  

!    REAL :: FACTOR         
!    REAL :: TQ1          
!    REAL :: TQ2         
!    REAL :: RQ1    
!    REAL :: RQ11 
!-------------------------------------------------    
    ! Use the Q10 relationship to determine the rates.
    ! Assume that TEMP is the maximum temperature.

!    TQ1  = 25.0
!    TQ2  = TEMP
!    FACTOR = LOG10( 2.0 ) * 0.1 * ( TQ1 - TQ2 )

!    RQ1  = K11    
!    RQ11 = LOG10( RQ1 ) - FACTOR
!    RQ11 = 10.0 ** RQ11

   ! VC&W Nitrification Rate
!    R_11   = RQ11 * NH4 * O2 

! 6/12/16 from GoMDOM code Where
!      nitmax =  .52 !(mmol m-3 d-1, maximum nitrification rate, calculated from
!Jenkins and Kemp 1984 and also the same value used in Pauer et al.  2016)
!!    KO2                      = 10.!(mmol O2 m-3, using the value for our KO2)
!      KNH4                     = 1. !71. !(mmol NH4 m-3, value from GoMDOM)
      R_11 = nitmax* O2/(KO2+O2) * NH4/(KNH4+NH4)


    RETURN
  END SUBROUTINE Nitrification
